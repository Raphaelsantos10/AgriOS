FARPHA Sprint 106.4 — instalação no Git Bash

cd ~
taskkill //F //IM node.exe 2>/dev/null || true
cd ~/Documents/Projetos
BACKUP="AgriOS_backup_sprint106_3_$(date +%Y%m%d_%H%M%S)"
mv AgriOS "$BACKUP"
unzip -o ~/Downloads/FARPHA_Sprint_106_4_Site_Institucional_Enterprise.zip -d ~/Documents/Projetos
cp "$BACKUP/frontend/.env.local" ~/Documents/Projetos/AgriOS/frontend/.env.local 2>/dev/null || true
cp -r "$BACKUP/.git" ~/Documents/Projetos/AgriOS/.git
cd ~/Documents/Projetos/AgriOS/frontend
npm install
rm -rf node_modules/.vite
npm run validate
npm run dev
