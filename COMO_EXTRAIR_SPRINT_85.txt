Feche VS Code e terminais dentro de AgriOS. Cole este bloco completo no Git Bash:

cd ~
taskkill //F //IM node.exe 2>/dev/null || true
cd ~/Documents/Projetos
BACKUP="AgriOS_backup_sprint84_$(date +%Y%m%d_%H%M%S)"
mv AgriOS "$BACKUP"
unzip -o ~/Downloads/FARPHA_Sprint_85_Dossie_Auditoria_PDF_ZIP.zip
cp "$BACKUP/frontend/.env.local" ~/Documents/Projetos/AgriOS/frontend/.env.local
cp -r "$BACKUP/.git" ~/Documents/Projetos/AgriOS/.git
cd ~/Documents/Projetos/AgriOS/frontend
npm install
npm run audit:prod
npm run validate
npm run preview
